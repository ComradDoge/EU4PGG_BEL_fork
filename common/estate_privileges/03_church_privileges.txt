			}

		}
		custom_tooltip = estate_church_monopoly_of_incense_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					trade_goods = incense
				}
				add_province_modifier = {
					name = estate_church_monopoly_of_incense_mod
					duration = -1
				}
			}
		}
	}
	on_revoked = {

		}
	}
	on_invalid = {
